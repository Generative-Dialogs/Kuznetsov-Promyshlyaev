65 def generate_image_response(self, prompt: str, target_path: str, character_images: Optional[List[str]] = None) -> Optional[str]:
66 """!
67 @brief Генерация изображения с использованием Gemini API
68
69 @param prompt Текстовый промпт для генерации изображения
70 @param target_path Полный путь для сохранения изображения
71 @param character_images Опциональный список путей к изображениям персонажей
72
73 @return Optional[str] Путь к сохраненному изображению или None в случае ошибки
74 """
75 try:
76 # Prepare contents list with prompt and images
77 contents: List[Any] = [prompt]
78
79 # Add character images if provided
80 if character_images:
81 for image_path in character_images:
82 try:
83 image = Image.open(image_path)
84 contents.append(image)
85 except Exception as e:
86 logging.warning(f"Failed to load character image {image_path}: {str(e)}")
87
88 # Generate the image using Gemini API
89 response = self.__client.models.generate_content(
90 model='gemini-2.0-flash-exp-image-generation',
91 contents=contents,
92 config=types.GenerateContentConfig(
93 response_modalities=['Text', 'Image']
94 )
95 )
96
97 # Process the response
98 for part in response.candidates[0].content.parts:
99 if part.inline_data is not None:
100 return self.__save_image(part.inline_data.data, target_path)
101
102 logging.error("No image data found in response")
103 return None
104
105 except Exception as e:
106 logging.error(f"Error generating image: {str(e)}")
107 return None
