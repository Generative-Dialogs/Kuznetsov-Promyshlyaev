120 def audio_to_text(self, audio_path: str) -> str:
121 """!
122 @brief Преобразование аудиофайла в текст
123
124 @param audio_path Путь к аудиофайлу для распознавания
125
126 @return str Распознанный текст или сообщение об ошибке
127
128 @throws sr.UnknownValueError если речь не может быть распознана
129 @throws sr.RequestError если возникла ошибка при подключении к API
130 """
131 temp_path = None
132 try:
133 # Добавляем тишину в начало файла
134 temp_path = self.__add_silence_padding(audio_path)
135
136 if self.__recognizer is None:
137 return "Ошибка: распознаватель речи не инициализирован"
138
139 with sr.AudioFile(temp_path) as source:
140 # Настройка распознавателя для уменьшения шума
141 self.__recognizer.adjust_for_ambient_noise(source, duration=0.5)
142 audio_data = self.__recognizer.record(source)
143
144 # Получаем код языка для Google Speech Recognition
145 language_code = self.__language_codes[self.__current_language]
146
147 # Сначала пробуем использовать Google Speech Recognition
148 try:
149 text = self.__recognizer.recognize_google(audio_data, language=language_code)
150 return str(text)
151 except (sr.UnknownValueError, sr.RequestError):
152 # Если Google не сработал, пробуем Sphinx
153 try:
154 text = self.__recognizer.recognize_sphinx(audio_data, language=language_code)
155 return str(text)
156 except (sr.UnknownValueError, sr.RequestError):
157 return "Речь не распознана"
158 except Exception as e:
159 return f"Ошибка при обработке аудио: {str(e)}"
160 finally:
161 # Удаляем временный файл
162 if temp_path:
163 self.__cleanup_temp_file(temp_path)
164
